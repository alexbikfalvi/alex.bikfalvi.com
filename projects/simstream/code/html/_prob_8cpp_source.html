00114 {
00115     // Calculate the cache index (do not go outside of cache boundary)
00116     if(ep > this->cacheEpMax) ep = this->cacheEpMax;

00118 
00119     if(ep < this->cacheEpMin) ep = this->cacheEpMin;
00120     if(ec < this->cacheEcMin) ec = this->cacheEcMin;
00121 
00122     unsigned int epi = (unsigned int)((ep - this->cacheEpMin)/this->cacheEpDelta);
00123     unsigned int eci = (unsigned int)((ec - this->cacheEcMin)/this->cacheEcDelta);
00124 
00125     assert(epi < this->cacheEpiMax);
00126     assert(eci < this->cacheEciMax);
00127 
00128     unsigned int cindex = eci * this->cacheEpiMax + epi;
00129 
00130     /*
00131      * Part I. If value is in the probability in the cache, return the cached value
00132      */
00133     if(this->cache[cindex] >= 0) return this->cache[cindex];
00134 
00135     /*
00136      * Part II. Else, calculate new value and insert it in the probability cache

00139     this->cache[cindex] = this->InvCdf0(epi, eci);
00140     this->mutex.Unlock();
