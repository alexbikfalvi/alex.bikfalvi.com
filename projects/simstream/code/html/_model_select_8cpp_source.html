00074 {
00075     // Select the optimal parent based on child remaining time : 
00076     // Search for the parent with the maximum remaining time less that the child remaining time 
00077     // and the minimum remaining time greater that the child remaining time
00078     assert(channel < this->numChannels);
00079     assert(child < this->numPeers);
00080 
00081     CPeer* peerChild = &this->peers[child];
00082 
00083     for(__uint32 layer = 0; layer < this->numLayers; layer++)
00084     {
00085         __time remainingGreater = 1E300;
00086         __time remainingLess = 0;
00087 
00088         CPeer* peerGreater = NULL;
00089         CPeer* peerLess = NULL;
00090 
00091         for(set<CPeer*>::iterator iter = this->index[channel].begin(); iter != this->index[channel].end(); iter++)
00092         {
00093             // Get the parent peer
00094             CPeer* peerParent = *iter;
00095             
00096             if(!this->IsResult(peerParent, layer)) // Check the peer was not selected parent for a previous layers
00097             {   
00098                 if(peerChild->VerifyParent(channel, layer, peerParent)) // Check the parent
00099                 {
00100                     if(peerParent->Remaining(time, channel) >= peerChild->Remaining(time, channel))
