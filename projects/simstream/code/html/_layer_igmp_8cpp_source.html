00057 void CLayerIgmp::Join(CAddress address, __uint32 entry)
00058 {
00059     __uint32 group = address.MulticastGroup();
00060     assert(group < this->numGroups);
00061 
00062     // If the host is part of the multicast group, return
00063     if(this->memberships[group].State() == CLayerIpMcastMembership::MEMBER) return;
00064 
00065     // Send an unsolicited membership report to the IP multicast address of the group
00066     CPacketIgmp* packet = new CPacketIgmp(address, CPacketIgmp::PACKET_IGMP_MEMBERSHIP_REPORT);
00067 
00068     // Send the IGMP packet on the joined interface, group multicast address and TTL of 1
00069     this->Send(entry, address, 1, packet);
00070     
00071     // Set the group state to joined
00072     this->memberships[group].Join(entry);

00079 {
00080     __uint32 group = address.MulticastGroup();
00081     assert(group < this->numGroups);
00082 
00083     // If the host is not part of the multicast group, return
00084     if(this->memberships[group].State() == CLayerIpMcastMembership::NON_MEMBER) return;
00085 
00086     // Send a membership leave to the all routers multicast address
00087     CPacketIgmp* packet = new CPacketIgmp(address, CPacketIgmp::PACKET_IGMP_LEAVE_GROUP);
